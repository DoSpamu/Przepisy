const SHEET_NAME = "Przepisy Community"; function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); // Zapis nowego przepisu przez GET ?action=add&... if(e.parameter.action === "add") { try { sheet.appendRow([ e.parameter.id || Date.now(), e.parameter.cat || "🥘 Wariacje", e.parameter.name || "", e.parameter.link || "", e.parameter.note || "community 🤝", e.parameter.q || "", e.parameter.skladniki|| "[]", new Date().toISOString() ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false, error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } // Odczyt wszystkich przepisów const rows = sheet.getDataRange().getValues(); if(rows.length <= 1) { return ContentService .createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } const headers = rows[0]; const data = rows.slice(1).map(row => { const obj = {}; headers.forEach((h,i) => { if(h === "skladniki") { try { obj[h] = JSON.parse(row[i]); } catch { obj[h] = []; } } else { obj[h] = row[i]; } }); return obj; }).filter(r => r.id); return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
